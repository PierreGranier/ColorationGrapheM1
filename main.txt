size_t FindIndex( const int a[], size_t size, int value )
{
    size_t index = 0;

    while ( index < size && a[index] != value ) ++index;

    return ( index == size ? -1 : index );
}

int indice_de(char* label) {
	int index = 0;
	while (index < NOMBRE_DE_SOMMETS) {
		if (SOMMETS[index].label == atoi(label))
			return index;
		++index;
	}
	return -1;
}
